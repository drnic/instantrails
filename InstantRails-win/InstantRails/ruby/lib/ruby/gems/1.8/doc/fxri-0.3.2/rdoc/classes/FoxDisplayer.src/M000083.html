        def display_method_info(method)
                init_text
                @formatter.draw_line(method.full_name)
                @formatter.display_params(method)
                @formatter.draw_line
                display_flow(method.comment)
                
                if method.aliases && !method.aliases.empty?
                        @formatter.blankline
                        aka = "(also known as "
                        aka << method.aliases.map {|a| a.name }.join(", ") 
                        aka << ")"
                        @formatter.wrap(aka)
                end
                set_text
        end
