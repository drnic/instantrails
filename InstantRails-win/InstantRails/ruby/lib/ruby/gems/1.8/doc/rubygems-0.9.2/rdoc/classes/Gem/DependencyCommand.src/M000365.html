    def find_reverse_dependencies(spec, srcindex)
      result = []
      srcindex.each do |name, sp|
        sp.dependencies.each do |dep|
          if spec.name == dep.name &&
              dep.version_requirements.satisfied_by?(spec.version)
            result << [sp.full_name, dep]
          end
        end
      end
      result
    end
