        def str_to_match_data(str, index=0)
                words = [ ]
                exclude = [ ]
                is_exclude = false
                
                while str[index]
                        case str[index]
                        when ?", ?'
                                word, index = get_word(str, index+1, str[index])
                                unless word.empty?
                                        if is_exclude
                                                exclude.push word
                                                is_exclude = false
                                        else
                                                words.push word
                                        end
                                end
                        
                        when 32 # space
                                is_exclude = false

??
                        when ?-
                                is_exclude = true
                        
                        else
                                word, index = get_word(str, index)
                                if is_exclude
                                        exclude.push word
                                        is_exclude = false
                                else
                                        words.push word
                                end                         
                        end
                        
                        index += 1
                end
                
                # check if word has upcase letters
                words.collect! do |w| 
                        [w, /[A-Z]/.match(w)!=nil]
                end
                exclude.collect! do |w| 
                        [w, /[A-Z]/.match(w)!=nil]
                end
                [words, exclude]
        end
