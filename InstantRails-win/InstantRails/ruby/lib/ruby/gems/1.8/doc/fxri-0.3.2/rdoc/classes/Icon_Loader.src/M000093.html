        def cfg_to_icons(cfg)
                cfg.attrs.each do |attr|
                        value = cfg.send(attr.to_sym)
                        if (value.class == Recursive_Open_Struct)
                                cfg_to_icons(value)
                        else
                                # value is a filename
                                icon = make_icon(value)
                                cfg.send((attr + "=").to_sym, icon)
                        end
                end
        end
