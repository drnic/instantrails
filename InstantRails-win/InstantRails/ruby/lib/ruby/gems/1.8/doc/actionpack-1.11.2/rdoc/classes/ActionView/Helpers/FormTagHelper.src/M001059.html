      def text_area_tag(name, content = nil, options = {})
        options = options.stringify_keys
        if options["size"]
          options["cols"], options["rows"] = options["size"].split("x")
          options.delete("size")
        end

        content_tag("textarea", content, { "name" => name, "id" => name }.update(options.stringify_keys))
      end
