                def change_case( answer_string )
                        if [:up, :upcase].include?(@case)
                                answer_string.upcase
                        elsif [:down, :downcase].include?(@case)
                                answer_string.downcase
                        elsif @case == :capitalize
                                answer_string.capitalize
                        else
                                answer_string
                        end
                end
