        def initialize(newParent, icon, *content)
                @content = content
                @sortable = Array.new(@content.size)
                @icon = icon
                @data = nil
                @parent = nil
                @sort_key = nil
                @searchable = nil
                # call parent=, don't forget the self!!
                self.parent = newParent
                show if newParent
                # update sortable
                @content.each_index do |pos|
                        update_sortable(pos, @content[pos]) if parent
                end
        end
