        def IRB.restart(im)
                irb = Irb.new(nil, im)    
        
                @CONF[:IRB_RC].call(irb.context) if @CONF[:IRB_RC]
                @CONF[:MAIN_CONTEXT] = irb.context
                trap("SIGINT") do
                        irb.signal_handle
                end

    class << irb.context.workspace.main
                        def gets
                                inp = IRB.conf[:MAIN_CONTEXT].io
                                inp.gets_mode = true
                                retval = IRB.conf[:MAIN_CONTEXT].io.gets
                                inp.gets_mode = false
                                retval
                        end
                end
    
    catch(:IRB_EXIT) do
      irb.eval_input
                end
    print "\n"

  end
