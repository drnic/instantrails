      def full?
         buf    = [0].pack('L') # dwFull
         needed = [0].pack('L')
         
         unless GetEventLogInformation(@handle, 0, buf, buf.size, needed)
            raise 'GetEventLogInformation() failed: ' + get_last_error
         end

         buf[0,4].unpack('L')[0] != 0
      end
