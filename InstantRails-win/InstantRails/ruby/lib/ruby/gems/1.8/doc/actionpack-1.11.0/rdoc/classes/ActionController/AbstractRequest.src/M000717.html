    def post_format
      @post_format ||=
           if env['HTTP_X_POST_DATA_FORMAT']
             env['HTTP_X_POST_DATA_FORMAT'].downcase.to_sym
           else
             case env['CONTENT_TYPE'].to_s.downcase
             when 'application/xml', 'text/xml'        then :xml
             when 'application/x-yaml', 'text/x-yaml'  then :yaml
             else :url_encoded
             end
           end
    end
