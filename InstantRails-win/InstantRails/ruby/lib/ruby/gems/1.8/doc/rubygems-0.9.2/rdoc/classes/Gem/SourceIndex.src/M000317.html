    def outdated
      remotes = Gem::SourceInfoCache.search(//)
      outdateds = []
      latest_specs.each do |_, local|
        name = local.name
        remote = remotes.select  { |spec| spec.name == name }.
                         sort_by { |spec| spec.version }.
                         last
        outdateds << name if remote and local.version < remote.version
      end
      outdateds
    end
