      def multi_to_wide(str)
         cp  = ($KCODE == 'UTF8') ? CP_UTF8 : CP_ACP
         buf = 0.chr * str.size * 2 # sizeof(WCHAR)
         int = MultiByteToWideChar(cp, 0, str, str.size, buf, buf.size)
         
         if int > 0
            buf[0, int*2]
         else
            str
         end
      end
