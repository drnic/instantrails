      def xml_http_request(path, parameters=nil, headers=nil)
        headers = (headers || {}).merge(
          "X-Requested-With" => "XMLHttpRequest",
          "Accept"           => "text/javascript, text/html, application/xml, text/xml, */*"
        )

        post(path, parameters, headers)
      end
