      def self.wave_volume
         volume = [0].pack('L')
         if @@waveOutGetVolume.call(-1, volume) != 0
            raise SoundError, get_last_error
         end
         volume = volume.unpack('L').first
         [low_word(volume), high_word(volume)]
      end
