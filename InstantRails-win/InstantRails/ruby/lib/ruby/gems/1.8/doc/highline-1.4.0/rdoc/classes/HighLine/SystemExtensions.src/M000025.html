      def terminal_size
        if /solaris/ =~ RUBY_PLATFORM and
           `stty` =~ /\brows = (\d+).*\bcolumns = (\d+)/
          [$2, $1].map { |c| x.to_i }
        else
          `stty size`.split.map { |x| x.to_i }.reverse
        end
      end
