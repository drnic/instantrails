    def initialize(*args)
      super()
      if args.length == 0
        @fm = FXGLPoint.new(-0.5, 0.0, 0.0)
        @to = FXGLPoint.new( 0.5, 0.0, 0.0)
      elsif args.length == 1
        @fm = args[0].fm
        @to = args[0].to
      else
        @fm = FXGLPoint.new(args[0], args[1], args[2])
        @to = FXGLPoint.new(args[3], args[4], args[5])
      end
    end
