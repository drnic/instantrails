        def encode_code( str, rs )
                str.gsub( %r{&}, '&amp;' ).
                        gsub( %r{<}, '&lt;' ).
                        gsub( %r{>}, '&gt;' ).
                        gsub( CodeEscapeRegexp ) {|match| EscapeTable[match][:md5]}
        end
