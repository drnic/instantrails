      def self.format_name(format_num)
         val = nil
         buf = 0.chr * 80
         self.open
         if GetClipboardFormatName(format_num, buf, buf.length) != 0
            val = buf
         end
         self.close
         val.split(0.chr).first rescue nil
      end
