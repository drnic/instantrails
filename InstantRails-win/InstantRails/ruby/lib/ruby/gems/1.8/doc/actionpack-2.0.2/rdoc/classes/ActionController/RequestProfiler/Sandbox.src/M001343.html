      def benchmark(n)
        @quiet = true
        print '  '
        result = Benchmark.realtime do
          n.times do |i|
            run
            print i % 10 == 0 ? 'x' : '.'
            $stdout.flush
          end
        end
        puts
        result
      ensure
        @quiet = false
      end
