  def list_fields(table, field=nil)
    command COM_FIELD_LIST, "#{table}\0#{field}", true
    f = read_rows 6
    fields = unpack_fields(f, @server_capabilities & CLIENT_LONG_FLAG != 0)
    res = Result::new self, fields, f.length
    res.eof = true
    res
  end
