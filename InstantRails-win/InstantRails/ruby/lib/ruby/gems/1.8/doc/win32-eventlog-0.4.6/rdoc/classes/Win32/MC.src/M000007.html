      def clean
         base = File.basename(@mc_file, '.mc')
         %w/.h .rc .res/.each do |ext|
            file = base + ext
            File.delete(file) if File.exists?(file)
         end
         Dir["MSG*.bin"].each do |binfile|
            File.delete(binfile)
         end
      end
