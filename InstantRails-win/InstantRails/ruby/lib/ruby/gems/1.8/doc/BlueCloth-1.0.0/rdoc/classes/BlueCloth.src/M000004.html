        def detab!( tabwidth=TabWidth )
                newstr = self.split( /\n/ ).collect {|line|
                        line.gsub( /(.*?)\t/ ) do
                                $1 + ' ' * (tabwidth - $1.length % tabwidth)
                        end
                }.join("\n")
                self.replace( newstr )
        end
