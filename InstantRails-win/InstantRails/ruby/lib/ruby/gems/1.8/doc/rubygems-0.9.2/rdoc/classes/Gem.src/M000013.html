    def configuration
      return @configuration if @configuration

      @configuration = {}
      class << @configuration
        undef_method :verbose # HACK RakeFileUtils pollution
      end if @configuration.respond_to? :verbose

      def @configuration.method_missing(sym, *args, &block)
        if args.empty?
          self[sym]
        else
          super
        end
      end

      @configuration
    end
