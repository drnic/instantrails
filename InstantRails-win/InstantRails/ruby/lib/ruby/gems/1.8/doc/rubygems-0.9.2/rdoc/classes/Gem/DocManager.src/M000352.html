    def load_rdoc
      if File.exist?(@doc_dir) && !File.writable?(@doc_dir)
        Gem::FilePermissionError.new(@doc_dir)
      end
      FileUtils.mkdir_p @doc_dir unless File.exist?(@doc_dir)
      begin
        require 'rdoc/rdoc'
      rescue LoadError => e
        raise DocumentError, 
          "ERROR: RDoc documentation generator not installed!"
      end
    end
