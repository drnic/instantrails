        def strip_link_definitions( str, rs )
                str.gsub( LinkRegex ) {|match|
                        id, url, title = $1, $2, $3

                        rs.urls[ id.downcase ] = encode_html( url )
                        unless title.nil?
                                rs.titles[ id.downcase ] = title.gsub( /"/, "&quot;" )
                        end
                        ""
                }
        end
