        def display_heading(text, level, indent)
                case level
                when 1
                        @proc.call(text, STYLE_H1)
                        @proc.call("\n")
                
                when 2
                        @proc.call(text, STYLE_H2)
                        @proc.call("\n")
                        
                else
                        @proc.call(indent)
                        @proc.call(text, STYLE_H3)
                        @proc.call("\n")
                end
        end
