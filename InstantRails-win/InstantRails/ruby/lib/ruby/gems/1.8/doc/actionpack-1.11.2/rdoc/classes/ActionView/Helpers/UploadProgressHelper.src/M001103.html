      def upload_status_progress_bar_tag(content='', options={})
        css = [options[:class], 'progressBar'].compact.join(' ')

        content_tag("div", 
          content_tag("div", 
            content_tag("div", 
              content_tag("div", content, :class => 'foreground'),
            :class => 'background'), 
          :class => 'border'), 
        {:id => progress_bar_id}.merge(options).merge({:class => css}))
      end
