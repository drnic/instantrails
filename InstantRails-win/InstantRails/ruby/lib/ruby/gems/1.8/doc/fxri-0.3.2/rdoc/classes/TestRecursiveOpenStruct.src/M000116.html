                def testKlammer
                        @s.a = "asdf"
                        assert_equal("asdf", @s["a"])
                        @s.b_x = "hog"
                        assert_equal("hog", @s["b_x"])
                        @s.c.b.a = 1234
                        assert_equal(1234, @s["c"]["b"]["a"])
                end
