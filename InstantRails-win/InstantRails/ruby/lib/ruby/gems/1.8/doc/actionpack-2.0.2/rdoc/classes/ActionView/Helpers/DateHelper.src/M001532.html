      def select_date(date = Date.today, options = {})
        options[:order] ||= []
        [:year, :month, :day].each { |o| options[:order].push(o) unless options[:order].include?(o) }

        select_date = ''
        options[:order].each do |o|
          select_date << self.send("select_#{o}", date, options)
        end
        select_date
      end
