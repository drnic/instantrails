        def transform_lists( str, rs )
                @log.debug " Transforming lists at %p" % (str[0,100] + '...')

                str.gsub( ListRegexp ) {|list|
                        @log.debug "  Found list %p" % list
                        bullet = $1
                        list_type = (ListMarkerUl.match(bullet) ? "ul" : "ol")
                        list.gsub!( /\n{2,}/, "\n\n\n" )

                        %{<%s>\n%s</%s>\n} % [
                                list_type,
                                transform_list_items( list, rs ),
                                list_type,
                        ]
                }
        end
