        def initialize(text_field)
                @text_field = text_field
                @formatter = FoxTextFormatter.new(70, "") do |arg, style|
                        startpos = @str.size
                        @str << arg
                        @formats.push [startpos, arg.size, style]                    
                end
                @reader = nil
        end
